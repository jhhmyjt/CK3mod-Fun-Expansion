#考验事件
hc.010 = {
	type = character_event
	title = hc.010.t
	desc = hc.010.desc
	left_portrait = {
		character = ROOT
		triggered_animation = {
			trigger = {
				prestige_level >= 3
				piety_level >= 3
			}
			animation = happiness
		}
		triggered_animation = {
			trigger = {
				NOT = {
					prestige_level >= 3
					piety_level >= 3
				}
			}
			animation = sadness
		}
	}
	theme = crown
	override_background = {
		reference = bedchamber
	}
	trigger = {
		is_ai = no
	}
	option = {
		name = {
			#good job
			trigger = {
				prestige_level >= 3
				piety_level >= 3
			}
			text = hc.010.a
		}
		name = {
			#bad
			trigger = {
				NOT = {
					prestige_level >= 3
					piety_level >= 3
				}
			}
			text = hc.010.b
		}
		if = {
			limit = {
				prestige_level >= 3
				piety_level >= 3
			}
			add_dynasty_prestige = 1000
			add_character_modifier = {
				modifier = hc_05_modifier
				days = 90
			}
		}
		else = {
			add_dynasty_prestige = -500
		}
	}
}
