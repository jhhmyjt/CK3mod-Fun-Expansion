#成功的
hc_call_war.002 = {
	type = character_event
	desc = hc_call_war.002.desc
	title = hc_call_war.002.t
	left_portrait = {
		character = root
		animation = personality_honorable
	}
	right_portrait = {
		character = scope:hc_call
		animation = personality_zealous
	}
	theme = war
	override_background = {
		reference = army_camp
	}
	trigger = {
		is_ai = no
	}
	immediate = {
		random_vassal = {
			limit = {
				AND = {
					has_trait = hc_knight
					is_at_war = no
				}
			}
			save_scope_as = hc_call
		}
	}
	option = {
		name = hc_call_war.002.a
		root = {
			every_character_war = {
				limit = {
					is_attacker = root
				}
				add_attacker = scope:hc_call
			}
			every_character_war = {
				limit = {
					is_defender = root
				}
				add_defender = scope:hc_call
			}
		}
		scope:hc_call = {
			add_prestige = 200
			add_character_modifier = hc_char_modifier_01
			trigger_event = {
				on_action = on_action_hc_knight
				months = 1
			}
		}
	}
}
