#成功的
hc_call_war.002 = {
	type = character_event
	desc = hc_call_war.002.desc
	title = hc_call_war.002.t
	left_portrait = {
		character = root
		animation = personality_honorable
	}
	right_portrait = {
		character = scope:hc_call
		animation = personality_zealous
	}
	theme = war
	override_background = {
		reference = army_camp
	}
	trigger = {
		is_ai = no
	}
	immediate = {
	}
	option = {
		name = hc_call_war.002.a
		random_vassal = {
			limit = {
				AND = {
					has_trait = hc_knight
					is_at_war = no
				}
			}
			save_scope_as = hc_call
		}
		every_character_war = {
			limit = {
				is_attacker = root
			}
			add_attacker = scope:hc_call
		}
		every_character_war = {
			limit = {
				is_defender = root
			}
			add_defender = scope:hc_call
		}
		scope:hc_call = {
			add_prestige = 200
			add_character_modifier = hc_char_modifier_01
			trigger_event = {
				on_action = on_action_hc_knight
				months = 1
			}
		}
	}
	option = {
		name = hc_call_war.002.b
		every_vassal = {
			limit = {
				has_trait = hc_knight
				is_at_war = no
			}
			save_scope_as = hc_called_knight
			add_prestige = 200
			add_character_modifier = hc_char_modifier_01
			trigger_event = {
				on_action = on_action_hc_knight
				months = 1
			}
			root = {
				every_character_war = {
					limit = {
						is_attacker = root
					}
					add_attacker = scope:hc_called_knight
				}
				every_character_war = {
					limit = {
						is_defender = root
					}
					add_defender = scope:hc_called_knight
				}
			}
		}
		while = {
			count = 3
			random_vassal = {
				limit = {
					is_at_war = no
					opinion = {
						target = scope:root
						value >= 99
					}
					not = {
						has_trait = hc_knight
					}
				}
				save_scope_as = hc_called_knight
				add_prestige = 200
				add_character_modifier = hc_char_modifier_01
				trigger_event = {
					on_action = on_action_hc_knight
					months = 1
				}
				root = {
					every_character_war = {
						limit = {
							is_attacker = root
						}
						add_attacker = scope:hc_called_knight
					}
					every_character_war = {
						limit = {
							is_defender = root
						}
						add_defender = scope:hc_called_knight
					}
				}
			}
		}
	}
}
